数组去除重复的项
```
let a = [1,'2',2,2,'3',3,3]

function unique(arr){
    return arr.filter((item,index,arr)=>{
        return  arr.indexOf(item) ===index
    })
}
console.log(unique(a))

function unique2(arr){
    return [...new Set(arr)]
}
console.log(unique2(a))

function unique3(arr){
    let resArray = []
    for(let item of arr){
        if(resArray.indexOf(item)==-1){
            resArray.push(item)
        }
    }
    return resArray
}
console.log(unique3(a))
```
